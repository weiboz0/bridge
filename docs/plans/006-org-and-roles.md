# Organization & Role System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the flat school/role model with a multi-tenant organization system where users register with no role and receive roles through OrgMembership records, enabling multi-org and multi-role support.

**Architecture:** Organizations become the tenant boundary, replacing the `schools` table. User roles are removed from the `users` table and moved to `orgMemberships`, allowing a user to hold different roles in different organizations. A `isPlatformAdmin` boolean on the user table handles platform-level admin access. Registration creates role-less users; org creation goes through a pending/approval workflow managed by platform admins.

**Tech Stack:** Drizzle ORM (pgTable, pgEnum), Next.js API routes, Zod v4 validation, Auth.js v5 (JWT callbacks), Vitest, bcryptjs

**Spec:** `docs/specs/002-platform-redesign.md` (Sub-project 1)

**Depends on:** Plans 001-005 (MVP complete)

/**
 * Create a NextRequest for testing API route handlers.
 */
export function createRequest(
  url: string,
  options: {
    method?: string;
    body?: unknown;
    searchParams?: Record<string, string>;
  } = {}
): NextRequest {
  const { method = "GET", body, searchParams } = options;

  const fullUrl = new URL(url, "http://localhost:3000");
  if (searchParams) {
    for (const [key, value] of Object.entries(searchParams)) {
      fullUrl.searchParams.set(key, value);
    }
  }

  const init: RequestInit = { method };
  if (body) {
    init.body = JSON.stringify(body);
    init.headers = { "Content-Type": "application/json" };
  }

  return new NextRequest(fullUrl, init);
}

/**
 * Parse JSON response body from a route handler Response.
 */
export async function parseResponse<T = unknown>(
  response: Response
): Promise<{ status: number; body: T }> {
  const status = response.status;
  const text = await response.text();
  try {
    const body = JSON.parse(text) as T;
    return { status, body };
  } catch {
    return { status, body: text as unknown as T };
  }
}

  db: Database,
  orgId: string,
  status: "pending" | "active" | "suspended"
) {
  const [existing] = await db
    .select()
    .from(organizations)
    .where(eq(organizations.id, orgId));

  if (!existing) return null;

  const updates: Partial<typeof organizations.$inferInsert> = {
    status,
    updatedAt: new Date(),
  };

  // Set verifiedAt when activating a school
  if (status === "active" && existing.type === "school" && !existing.verifiedAt) {
    updates.verifiedAt = new Date();
  }

  const [updated] = await db
    .update(organizations)
    .set(updates)
    .where(eq(organizations.id, orgId))
    .returning();
  return updated || null;
}

export async function listOrgMembers(
  db: Database,
  orgId: string,
  filters?: { role?: "org_admin" | "teacher" | "student" | "parent" }
) {
  const conditions = [eq(orgMemberships.orgId, orgId)];
  if (filters?.role) {
    conditions.push(eq(orgMemberships.role, filters.role));
  }

  return db

      userId: orgMemberships.userId,
      role: orgMemberships.role,
      status: orgMemberships.status,

      userName: users.name,
      userEmail: users.email,

}

import { NextRequest, NextResponse } from "next/server";
import { z } from "zod";
import bcrypt from "bcryptjs";
import { db } from "@/lib/db";
import { users, authProviders } from "@/lib/db/schema";
import { eq } from "drizzle-orm";

const registerSchema = z.object({
  name: z.string().min(1).max(255),
  email: z.string().email(),
  password: z.string().min(8).max(128),
});

export async function POST(request: NextRequest) {
  const body = await request.json();
  const parsed = registerSchema.safeParse(body);

  if (!parsed.success) {
    return NextResponse.json(
      { error: "Invalid input", details: parsed.error.flatten() },
      { status: 400 }
    );
  }

  const { name, email, password } = parsed.data;

  const [existing] = await db
    .select()
    .from(users)
    .where(eq(users.email, email));

  if (existing) {
    return NextResponse.json(
      { error: "Email already registered" },
      { status: 409 }
    );
  }

  const passwordHash = await bcrypt.hash(password, 10);

  const [user] = await db
    .insert(users)
    .values({
      name,
      email,
      passwordHash,
    })
    .returning({

import { NextRequest, NextResponse } from "next/server";
import { z } from "zod";
import { auth } from "@/lib/auth";
import { db } from "@/lib/db";
import { createOrganization, getOrganizationBySlug } from "@/lib/organizations";
import { addOrgMember, getUserOrgMemberships } from "@/lib/org-memberships";

const createOrgSchema = z.object({
  name: z.string().min(1).max(255),
  slug: z
    .string()
    .min(1)
    .max(255)
    .regex(/^[a-z0-9]+(?:-[a-z0-9]+)*$/, "Slug must be lowercase alphanumeric with hyphens"),
  type: z.enum(["school", "tutoring_center", "bootcamp", "other"]),
  contactEmail: z.string().email(),
  contactName: z.string().min(1).max(255),
  domain: z.string().max(255).optional(),
});

export async function POST(request: NextRequest) {
  const session = await auth();
  if (!session?.user?.id) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const body = await request.json();
  const parsed = createOrgSchema.safeParse(body);

  if (!parsed.success) {
    return NextResponse.json(
      { error: "Invalid input", details: parsed.error.flatten() },
      { status: 400 }
    );
  }

  // Check slug uniqueness
  const existing = await getOrganizationBySlug(db, parsed.data.slug);
  if (existing) {
    return NextResponse.json(
      { error: "An organization with this slug already exists" },
      { status: 409 }
    );
  }

  const org = await createOrganization(db, parsed.data);

  // Make the creator an org_admin
  const membership = await addOrgMember(db, {
    orgId: org.id,
    userId: session.user.id,
    role: "org_admin",
    status: "active",
  });

  return NextResponse.json({ org, membership }, { status: 201 });
}

export async function GET() {
  const session = await auth();
  if (!session?.user?.id) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const memberships = await getUserOrgMemberships(db, session.user.id);
  return NextResponse.json(memberships);
}

```ts
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@/lib/auth";
import { db } from "@/lib/db";
import { getOrganization } from "@/lib/organizations";
import { getUserOrgMemberships } from "@/lib/org-memberships";

export async function GET(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  const session = await auth();
  if (!session?.user?.id) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const { id } = await params;
  const org = await getOrganization(db, id);

  if (!org) {
    return NextResponse.json({ error: "Organization not found" }, { status: 404 });
  }

  // Check membership unless platform admin
  if (!session.user.isPlatformAdmin) {
    const memberships = await getUserOrgMemberships(db, session.user.id);
    const isMember = memberships.some((m) => m.orgId === id);
    if (!isMember) {
      return NextResponse.json(
        { error: "You are not a member of this organization" },
        { status: 403 }
      );
    }
  }

  return NextResponse.json(org);
}

  const memberships = await getUserOrgMemberships(db, userId);

export async function POST(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  const session = await auth();
  if (!session?.user?.id) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const { id: orgId } = await params;

  // Only org admins or platform admins can add members
  if (!session.user.isPlatformAdmin && !(await isOrgAdmin(session.user.id, orgId))) {
    return NextResponse.json(
      { error: "Only org admins can add members" },
      { status: 403 }
    );
  }

  const body = await request.json();
  const parsed = addMemberSchema.safeParse(body);

  if (!parsed.success) {
    return NextResponse.json(
      { error: "Invalid input", details: parsed.error.flatten() },
      { status: 400 }
    );
  }

  // Find the user by email
  const [targetUser] = await db
    .select()
    .from(users)
    .where(eq(users.email, parsed.data.email));

  if (!targetUser) {
    return NextResponse.json(
      { error: "No user found with this email" },
      { status: 404 }
    );
  }

  try {
    const membership = await addOrgMember(db, {
      orgId,
      userId: targetUser.id,
      role: parsed.data.role,
      invitedBy: session.user.id,
    });
    return NextResponse.json(membership, { status: 201 });
  } catch (error: any) {
    if (error.message?.includes("unique") || error.code === "23505") {
      return NextResponse.json(
        { error: "User already has this role in the organization" },
        { status: 409 }
      );
    }
    throw error;
  }
}

async function isOrgAdmin(userId: string, orgId: string): Promise<boolean> {

export async function PATCH(
  request: NextRequest,
  { params }: { params: Promise<{ id: string; memberId: string }> }
) {
  const session = await auth();
  if (!session?.user?.id) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const { id: orgId, memberId } = await params;

  if (!session.user.isPlatformAdmin && !(await isOrgAdmin(session.user.id, orgId))) {

      { error: "Only org admins can update members" },
      { status: 403 }
    );

  const parsed = updateSchema.safeParse(body);

  if (!parsed.success) {
    return NextResponse.json(
      { error: "Invalid input", details: parsed.error.flatten() },
      { status: 400 }
    );
  }

  let result = null;

  if (parsed.data.status) {
    result = await updateMembershipStatus(db, memberId, parsed.data.status);
  }

  if (parsed.data.role) {
    result = await updateMembershipRole(db, memberId, parsed.data.role);
  }

  if (!result) {
    return NextResponse.json(
      { error: "Membership not found" },
      { status: 404 }
    );
  }

  return NextResponse.json(result);
}

  request: NextRequest,

  const session = await auth();
  if (!session?.user?.id) {

/**
 * Data migration: schools → organizations, users.role → orgMemberships
 *
 * Run this BEFORE applying the Drizzle schema migration that drops
 * the schools table and removes role/schoolId from users.
 *
 * Usage:
 *   DATABASE_URL="postgresql://..." bun run scripts/migrate-to-orgs.ts
 *
 * This script is idempotent — safe to run multiple times.
 */
import postgres from "postgres";
import { drizzle } from "drizzle-orm/postgres-js";

const connectionString = process.env.DATABASE_URL;
if (!connectionString) {
  console.error("DATABASE_URL is required");
  process.exit(1);
}

const client = postgres(connectionString);
const db = drizzle(client);

async function migrate() {
  console.log("Starting data migration...");

  // Step 1: Check if old tables/columns still exist
  const [schoolsExist] = await client`
    SELECT EXISTS (
      SELECT 1 FROM information_schema.tables
      WHERE table_name = 'schools'
    ) as exists
  `;

  if (!schoolsExist.exists) {
    console.log("No schools table found — skipping school migration.");
  } else {
    // Step 2: Migrate schools → organizations
    const schools = await client`SELECT * FROM schools`;
    console.log(`Found ${schools.length} schools to migrate.`);

    for (const school of schools) {
      // Check if already migrated (by matching name)
      const [existing] = await client`
        SELECT id FROM organizations WHERE name = ${school.name}
      `;

      if (existing) {
        console.log(`  Skipping "${school.name}" — already migrated.`);
        continue;
      }

      // Create a slug from the school name
      const slug = school.name
        .toLowerCase()
        .replace(/[^a-z0-9]+/g, "-")
        .replace(/^-|-$/g, "");

      await client`
        INSERT INTO organizations (id, name, slug, type, status, contact_email, contact_name, settings, created_at, updated_at)
        VALUES (
          ${school.id},
          ${school.name},
          ${slug},
          'school',
          'active',
          'admin@placeholder.com',
          'Admin',
          ${JSON.stringify(school.settings || {})},
          ${school.created_at},
          ${school.updated_at}
        )
      `;
      console.log(`  Migrated school "${school.name}" → org "${slug}".`);
    }
  }

  // Step 3: Migrate users.role → orgMemberships
  const [roleColumnExists] = await client`
    SELECT EXISTS (
      SELECT 1 FROM information_schema.columns
      WHERE table_name = 'users' AND column_name = 'role'
    ) as exists
  `;

  if (!roleColumnExists.exists) {
    console.log("No role column on users — skipping role migration.");
  } else {
    const usersWithRoles = await client`
      SELECT id, role, school_id FROM users WHERE role IS NOT NULL
    `;
    console.log(`Found ${usersWithRoles.length} users with roles to migrate.`);

    for (const user of usersWithRoles) {
      // Skip admin role — those become isPlatformAdmin
      if (user.role === "admin") {
        await client`
          UPDATE users SET is_platform_admin = true WHERE id = ${user.id}
        `;
        console.log(`  Set isPlatformAdmin for user ${user.id}.`);
        continue;
      }

      // If user has a schoolId, create membership in that org
      if (user.school_id) {
        const [existingMembership] = await client`
          SELECT id FROM org_memberships
          WHERE org_id = ${user.school_id}
          AND user_id = ${user.id}
          AND role = ${user.role}
        `;

        if (!existingMembership) {
          await client`
            INSERT INTO org_memberships (org_id, user_id, role, status, created_at)
            VALUES (${user.school_id}, ${user.id}, ${user.role}, 'active', NOW())
          `;
          console.log(`  Created ${user.role} membership for user ${user.id} in org ${user.school_id}.`);
        } else {
          console.log(`  Skipping — membership already exists for user ${user.id}.`);
        }
      } else {
        console.log(`  Skipping user ${user.id} (${user.role}) — no school_id.`);
      }
    }
  }

  console.log("Migration complete.");
  await client.end();
}

migrate().catch((err) => {
  console.error("Migration failed:", err);
  process.exit(1);
});
```
